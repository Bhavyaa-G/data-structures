#include<stdio.h>
int stack[100],choice,n,top,x,i,beg,end;
void push(void);
void pop(void);
void display(void);
void swap(int a, int b)
{
    int temp = stack[a];
    stack[a] = stack[b];
    stack[b] = temp;
}
void push()
{
 if(top>=n-1)
 {
 printf("\n\tSTACK is over flow");
 
 }
 else
 {
 printf(" Enter a value to be pushed:");
 scanf("%d",&x);
 top++;
 stack[top]=x;
 }
}
void pop()
{
 if(top<=-1)
 {
 printf("\n\t Stack is under flow");
 }
 else
 {
 printf("\n\t The popped elements is %d",stack[top]);
 top--;
 }
}
void Reverse(int front,int rear){
    if (front<rear) //base condition
    { 
    	swap(front,rear);
    	Reverse(front+1,rear-1); //recursion
    }
}
void display()
{
 if(top>=0)
 {
 printf("\n The elements in STACK \n");
 for(i=top; i>=0; i--)
     printf("\n%d",stack[i]);
 printf("\n Press Next Choice");
 }
 else
 {
 printf("\n The STACK is empty");
 }
 
}
int main()
{
 top=-1;
 printf("\n Enter the size of STACK[MAX=100]:");
 scanf("%d",&n);
  beg=top;
  end=n-1;
 printf("\n\t STACK OPERATIONS USING ARRAY");
 printf("\n\t--------------------------------");
 printf("\n\t 1.PUSH\n\t 2.POP\n\t 3.REVERSE\n\t 4.DISPLAY\n\t 5.EXIT\n\t ");
 do
 {
 printf("\n Enter the Choice:");
 scanf("%d",&choice);
 switch(choice)
{
 case 1:
 {
 push();
 break;
 }
 case 2:
 {
 pop();
 break;
 }
 case 3:
 {
 Reverse(beg,end);
 printf("\n The array is Reversed");
 printf("\n Press Next Choice \n");
 break;
 }
 case 4:
 {
 display();
 break;
 }
 case 5:
 {
 printf("\n\t EXIT POINT ");
 break;
 }
 default:
 {
 printf ("\n\t Please Enter a Valid Choice(1/2/3/4/5)");
 }
 
 }
 }
 while(choice!=5);
 return 0;
}
